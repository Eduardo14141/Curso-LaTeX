			f(x) = \sum_{n=0}^{\infty}{
				\dfrac{f^{(n)}(a)}{n!}(x-a)^{n}

		\end{equation}

			\sinh(x) &= \sum_{n=0}^{\infty}{\dfrac{x^{2n+1}}{(2n+1)!}} &

		\end{align}
